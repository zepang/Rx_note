# 栈数据结构

栈是一种遵循先进后出原则的有序集合。

## 基于数组的栈

```js
class Stack {
  constructor () {
    this.items = []
  }

  // 添加元素

    this.items.push(element)
  }

  // 栈是否为空
```

## 基于JavaScript对象的栈

```js
class Stack {
  constructor () {
    this.count = 0
    this.items = {}
  }

  push (element) {
    this.items[this.count++] = element
  }

  pop () {
    if (this.isEmpty) {
      return undefined
    }

    this.count--
    let result = this.items[this.count]
    delete this.items[this.count]
    return result
  }

  isEmpty () {
    return this.count === 0
  }

  size () {
    return this.count
  }

  peek () {
    if (this.isEmpty) {
      return undefined
    }

    return this.items[this.count - 1]
  }

  clear () {
    this.items = {}
    this.count = 0
  }

  toString() { 
    if (this.isEmpty()) { 
      return '';
    } 

    let objString = `${this.items[0]}`
    for (let i = 1; i < this.count; i++) {  
      objString = `${objString},${this.items[i]}`  
    } 
    return objString
  }
}
```


# 队列

```js
  }

    return this.count - this.lowestCount === 0
  }

  size () {
    return this.count - this.lowestCount
  }

    this.items = {}
    this.count = 0
    this.lowestCount = 0
  }

  toString () {
    if (this.isEmpty) {
      return ''
    }

    let objString = this.items[this.lowestCount]
    for (let i = this.lowestCount; i < this.count; i++) {
      objString = `${objString}, ${this.items[i]}`
    }

    return objString
  }
}
```
