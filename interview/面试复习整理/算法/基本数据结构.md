# 栈数据结构

栈是一种遵循先进后出原则的有序集合。

## 基于数组的栈

```js
class Stack {
  constructor () {
    this.items = []
  }

  // 添加元素
  push (element) {
    this.items.push(element)
  }

  // 移除元素
  pop () {
    return this.items.pop()
  }

  // 返回栈顶元素
  peek () {
    return this.items[this.item.length - 1]
  }
  
  // 栈是否为空
  isEmpty () {
    return this.items.length === 0
  }

  // 栈大小
  size () {
    return this.items.length
  }

  // 清空栈
  clear () {
    this.items = []
  }
}
```

## 基于JavaScript对象的栈

```js
class Stack {
  constructor () {
    this.count = 0
    this.items = {}
  }

  push (element) {
    this.items[this.count++] = element
  }

  pop () {
    if (this.isEmpty) {
      return undefined
    }

    this.count--
    let result = this.items[this.count]
    delete this.items[this.count]
    return result
  }

  isEmpty () {
    return this.count === 0
  }

  size () {
    return this.count
  }

  peek () {
    if (this.isEmpty) {
      return undefined
    }

    return this.items[this.count - 1]
  }

  clear () {
    this.items = {}
    this.count = 0
  }

  toString() { 
    if (this.isEmpty()) { 
      return '';
    } 

    let objString = `${this.items[0]}`
    for (let i = 1; i < this.count; i++) {  
      objString = `${objString},${this.items[i]}`  
    } 
    return objString
  }
}
```

# 双端队列

双端队列是一种允许我们从前端和后端同时增加和删除元素的队列。

```js
class Deque {
  constructor () {
    this.count = 0
    this.lowestCount = 0
    this.items = {}
  } 

  addFront (element) {
    if (this.isEmpty) {
      this.addBack(element)
    } else if (this.lowestCount > 0) {
      this.lowestCount--
      this.items[this.lowestCount] = element
    } else {
      for (let i = this.count; i > 0; i--) {
        // 后移
        this.items[i] = this.items[i - 1]
      }
      this.count++
      this.lowestCount = 0
      this.items[0] = element
    }
  }

  addBack (element) {
    this.items[this.count++] = element
  }

  removeFront () {
    if (this.isEmpty) {
      return undefined
    }
    let result = this.items[this.lowestCount]
    delete this.items[this.lowestCount]
    this.lowestCount++
    return 
  }

  removeBack () {
    if (this.isEmpty) {
      return undefined
    }

    this.count--
    let result = this.items[this.count]
    delete this.items[this.count]
    return result

      return undefined
    }

    return this.items[this.lowestCount]
```

# 链表

要存储多个元素，数组是可能是最常用的数据结构，但是，数组的大小是固定的，要从起始或者中间位置插入或者移除元素的成本比较高，因为需要移动元素。
链表存储有序的元素集合，不同于数组，链表中的元素在内存中不是连续放置的，每一个元素由一个存储元素本身的节点和一个指向下一个元素的引用来组成。

```js
function equal (a, b) {
  return a === b
}

class Node {
  constructor (element) {
    this.element = element
    this.next = null
  }
}

class LinkList {
  constructor () {
    this.head = null
    this.count = 0
  }

  push (element) {
    let node = new Node(element)
    if (this.head === null) {
      this.head = node
    } else {
      let current = head
      // 找到链表中最后一项
      while(current.next !== null) {
        current = current.next
      }

      current.next = node
    }
    this.count++
  }

  getElementAt (index) {
    if (index >= 0 && index < this.count) {
      let node = head
      for (let i = 0; i < index && node !== null; i++) {
        node = node.next
      }

      return node
    }
    return undefined
  }


  insert (element, index) {
    if (index >= 0 && index < this.count) {
      const node = new Node(element)
      if (index === 0) {
        const current = this.head
        node.next = current
        this.head = node
      } else {
        let previous = this.getElementAt(index - 1)
        let current = this.getElementAt(index)
        node.next = current
        previous.next = node
      }
      this.count++
      return true
    }
    return false
  }

  removeAr (index) {
    if (index >= 0 && index < this.count) {
      if (index === 0) {
        this.heade = this.head.next
      } else {
        let previous = this.getElementAt(index - 1)
        let current = this.getElementAt(index)
        // 通过把previous的next和current的next连接起来，跳过current，达到移除效果
        previous.next = current.next
      }
      this.count--
      return current.element
    }
    return undefined
  }

  indexOf (element) {
    let current = this.head
    for (let i = 0; i < this.count; i++) {
      if (equal(current.element, element)) {
        return i
      }

      current = current.next
    }
    return -1
  }

  size () {
    return this.count
  }

  isEmpty () {
    return this.count === 0
  }

  getHead () {
    if (this.head !== null) return this.head
    return undefined
  }

  toString () {
    if (this.head === null) {
      return ''
    }

    let objString = `${this.head.element}`
    let current = this.head
    for (let i = 0; i < this.size() && current !== null; i++) {
      objString = `${objString}, ${current.element}`
    }

    return objString
  }
}
```
